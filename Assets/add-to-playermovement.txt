    private Collider2D playerCollider;

    void Start()
    {
        rb = GetComponent<Rigidbody2D>();
        playerCollider = GetComponent<Collider2D>();
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.S))
        {
            StartCoroutine(DropDown());
        }
    }

    private IEnumerator DropDown()
    {
        // Temporarily switch layer to avoid collision with platform
        gameObject.layer = LayerMask.NameToLayer("Ignore Raycast");
        playerCollider.enabled = false;
        yield return new WaitForSeconds(0.5f); // Adjust time as needed
        playerCollider.enabled = true;
        gameObject.layer = LayerMask.NameToLayer("Default"); // Or your original layer
    }
